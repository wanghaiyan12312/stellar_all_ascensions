		limit = { has_country_flag = gray_army_active }

		class = official

		species = root

			1 = leader_trait_governor_gray

			set_leader_flag = immune_to_negative_traits

		}
	}
	last_created_leader = {
		save_global_event_target_as = gray_official
		set_leader_flag = has_gained_level_trait
	}
	if = {
		limit = {
			has_country_flag = gray_shroud_update
		}
		event_target:gray_official = {
			update_legendary_govern_official = yes
			add_trait = {
				trait = leader_trait_governor_gray
				show_message = no
				consume_selection = yes
			}
		}
	}
}
