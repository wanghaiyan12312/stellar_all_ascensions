titan_ships_limit = {
	ship_types = {
		titan
		cosmo_crisis_titan
		bio_titan
		cosmo_crisis_stinger
	}

	base = 16 # 1 * titan's size_multiplier
	max = 8000 # 500 * 16
	naval_cap_fraction = 1

	show = {
		is_scope_valid = yes
		OR = {
			has_technology = tech_titans
			has_technology = tech_cosmogenesis_FE_titan
			has_technology = tech_cosmogenesis_stinger
		}
	}
}
